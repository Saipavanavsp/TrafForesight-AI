\section{Source Code}

\subsection{Neural Network Model}
\paragraph{} The following code defines the shared actor-critic policy network used by the PPO agent. It takes a seven-dimensional observation as input and produces action logits along with a state value estimate.

\begin{verbatim}
class MLPPolicy3D(nn.Module):
    def __init__(self, obs_dim=7, act_dim=5, hidden=256):
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(obs_dim, hidden), nn.Tanh(),
            nn.Linear(hidden, hidden), nn.Tanh(),
            nn.Linear(hidden, hidden // 2), nn.Tanh(),
        )
        self.actor  = nn.Linear(hidden // 2, act_dim)
        self.critic = nn.Linear(hidden // 2, 1)

    def forward(self, x):
        h = self.net(x)
        return self.actor(h), self.critic(h).squeeze(-1)
\end{verbatim}

\subsection{Player Mechanics}
\paragraph{} The Player class handles the spaceship's health, shield, shot tracking, and movement based on keyboard input.

\begin{verbatim}
class Player:
    def __init__(self):
        self.health = 100.0
        self.shield = 50.0
        self.shots_fired = 0
        self.shots_hit = 0

    def update(self, dt, keys):
        if keys[pygame.K_LEFT]:
            self.x -= self.speed * dt
        if keys[pygame.K_RIGHT]:
            self.x += self.speed * dt
\end{verbatim}

\subsection{Dynamic Difficulty Adjustment}
\paragraph{} This snippet shows how player performance metrics are collected, normalized into an observation vector, and fed to the trained PPO model. The model's output action is then used to adjust enemy count and speed.

\begin{verbatim}
obs_array = np.array([

obs_tensor = torch.as_tensor(obs_array, dtype=torch.float32).unsqueeze(0)

logits, _ = self.agent(obs_tensor)
action = torch.argmax(logits, dim=-1).item()

\subsection{Game Loop}
\paragraph{} The main game loop runs at a fixed frame rate, updating the player, spawning lasers on spacebar press, updating all enemies, and triggering the DDA agent at regular intervals.

\begin{verbatim}
while running:
    dt = clock.tick(FPS) / 1000.0

    player.update(dt, keys)

    if keys[pygame.K_SPACE]:
        lasers.append(Laser(...))

    for enemy in enemies:
        enemy.update(dt, player)

    if agent_timer >= 2.0:
        agent_step(player)
\end{verbatim}
